async function fetchStock(input){ const ticker = input.value.trim().toUpperCase(); if(!ticker) return; const row = input.closest("tr"); try{ const res = await fetch( `https://query1.finance.yahoo.com/v10/finance/quoteSummary/${ticker}?modules=price,summaryDetail` ); const data = await res.json(); const result = data.quoteSummary.result[0]; const priceData = result.price || {}; const detail = result.summaryDetail || {}; const currentPrice = priceData.regularMarketPrice?.raw || 0; const prevClose = detail.previousClose?.raw || 0; const changeAmount = currentPrice - prevClose; const changePercent = prevClose ? (changeAmount / prevClose) * 100 : 0; row.querySelector(".stockName") .innerText = priceData.longName || "-"; row.querySelector(".exchange") .innerText = priceData.exchangeName || "-"; row.querySelector(".price") .innerText = "$" + currentPrice.toFixed(2); const changePercentEl = row.querySelector(".changePercent"); const changeAmountEl = row.querySelector(".changeAmount"); changePercentEl.innerText = changePercent.toFixed(2) + "%"; changeAmountEl.innerText = changeAmount.toFixed(2); if(changePercent < 0){ changePercentEl.classList.add("down"); changeAmountEl.classList.add("down"); }else{ changePercentEl.classList.remove("down"); changeAmountEl.classList.remove("down"); } const yieldValue = detail.dividendYield?.raw ? detail.dividendYield.raw * 100 : 0; if(yieldValue){ row.querySelector(".yield") .innerText = yieldValue.toFixed(2) + "%"; }else{ row.querySelector(".yield") .innerHTML = `
%
`; } const rate = detail.dividendRate?.raw || 0; if(rate){ row.querySelector(".rate") .innerText = rate.toFixed(2); }else{ row.querySelector(".rate") .innerHTML = ` `; } let freq = "-"; if(detail.dividendRate){ if( detail.trailingAnnualDividendRate?.raw ){ freq = "Quarterly"; } } row.querySelector(".freq") .innerText = freq; row.dataset.dividendYield = yieldValue || 0; updatePortfolio(row); }catch(err){ console.log(err); row.querySelector(".exchange") .innerText = "YAHOO ERROR"; } }
